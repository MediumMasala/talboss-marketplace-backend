-- Adds marketplace_tal and marketplace_round1 breakdown to the daily RPC.
-- "both"-source rows count toward both buckets so the parts >= marketplace.

create or replace function daily_breakdown(start_date date, end_date date)
returns table (
  joined_at date,
  total bigint,
  marketplace bigint,
  marketplace_tal bigint,
  marketplace_round1 bigint,
  tier1_supreme bigint,
  tal_users bigint,
  round1 bigint
)
language sql
stable
as $$
  select
    joined_at,
    count(*) as total,
    count(*) filter (
      where (is_marketplace = true and confidence = 'high')
         or confidence = 'medium'
    ) as marketplace,
    count(*) filter (
      where ((is_marketplace = true and confidence = 'high')
             or confidence = 'medium')
        and source_table in ('tal_users','both')
    ) as marketplace_tal,
    count(*) filter (
      where ((is_marketplace = true and confidence = 'high')
             or confidence = 'medium')
        and source_table in ('round1_god_table','both')
    ) as marketplace_round1,
    count(*) filter (
      where is_marketplace = true
        and tier in ('tier1','supreme')
    ) as tier1_supreme,
    count(*) filter (where source_table in ('tal_users','both')) as tal_users,
    count(*) filter (where source_table in ('round1_god_table','both')) as round1
  from candidates_daily
  where joined_at between start_date and end_date
  group by joined_at
  order by joined_at;
$$;
